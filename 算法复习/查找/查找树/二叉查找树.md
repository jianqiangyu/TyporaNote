二叉查找树

二叉查找树是一棵特殊的二叉树，二叉树中的关键字总是以满足二叉搜索树性质的方式来存储：

> 设`x`是二叉搜索中的一个结点。如果`y`是`x`的左子树中的一个节点，那么`y.key <= x.key`；如果`y`是`x`的右子树中的一个节点，那么`y.key >= x.key`;

```c
struct TreeNode{
	struct TreeNode *right;
    struct TreeNode *left;
    int val;
}

struct TreeNode *minimum(struct TreeNode * root) {
	
    if(root == NULL) {
        return NULL;
    }
    
    while(root->left) {
        root=root->left;
    }
    return root;
}


struct TreeNode *maximum(struct TreeNode * root) {
	
    if(root == NULL) {
        return NULL;
    }
    
    while(root->right) {
        root=root->right;
    }
    return root;
}

struct TreeNode *search(struct TreeNode * root, int key) {
    
    if(root == NULL) {
        return NULL;
    }else if(key < root->val ) {
        return search(root->left,key);
    }else if(key > root->val){
        return search(root->right,key);
    }else {
        return root;
    }
}

struct TreeNode *createNode(int key) {
     struct TreeNode *node =  malloc(sizeof(struct TreeNode));
        if(node == NULL) {
            return NULL;
        }
        node->val = key;
        node->left = NULL;
        node->right = NULL;
        return node;
}

struct TreeNode *insert(struct TreeNode * root, int key) {

    struct TreeNode * pre = NULL;
    while(root!=nil) {
        if(key <= root->val) {
            root = root->left;
        }else{
            root = root->right;
        }
        pre = root;
    }
    
	if(pre == NULL) {
    	//根节点
	    return create(key);
    }else if(key <= pre->val) {
		pre->left = create(key);
    }else {
        pre->right = create(key);
    }
    return root;
}

struct TreeNode *delete(struct TreeNode * root, int key) {

    if(root == NULL) {
        return NULL;
    }else if(key < root->val){
		root->left = delete(root->left, key);
    }else if(key > root->val) {
        root->right = delete(root->right, key);
    }else if(root->left != NULL && root->right != NULL){
        struct TreeNode *originRoot = root;
		root = minimum(root->right);
        root->right = delete(originRoot->right);
        root->left = originRoot->left;
    }else {
        struct TreeNode *temp = root;
        if(root->left == NULL) {
            root = root->right;
        }
        if(root->right == NULL) {
            root = root->left;
        }
        free(temp);
    }
    
    return root;
}
```
